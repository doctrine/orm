  271.

  273.         
  274.         $ret = array();
  275.         if ( ! empty($terms)) {
  276.             foreach ($terms as $i => $term) {
  277.                 if (empty($term)) {
  278.                     continue;
  279.                 }
  280.                 $lower = strtolower(trim($term));
  281.  
  282.                 if (in_array($lower, self::$_stopwords)) {
  283.                     continue;
  284.                 }
  285.  
  286.                 $ret[$i] = $lower;
  287.             }
  288.         }
  289.         return $ret;
  290.     }
